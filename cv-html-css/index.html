function maxSum(root) {
    let max = 0;
    const byPass = (node) => {
            if (node === undefined || node === null) return 0;
            const maxLeft = byPass(node.left);
            const maxRight = byPass(node.right);
            const maxPreConnected = Math.max(node.value + maxLeft, node.value + maxRight);
            const maxConnected = Math.max(maxPreConnected, maxLeft + node.value, maxRight + node.value);
            max = Math.max(maxLeft, maxRight, maxConnected, node.value);
            return maxConnected;
    }
    byPass(root);
    return max;
}
